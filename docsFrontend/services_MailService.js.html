/**
 * Module for reset a password
 * @module MailService
 */
import axios from "axios";

/**
 * Send an email with an URL to change password
 * @method forgotPassword
 * @param {*} e Event
 * @param {Bool} setValidated Validate data form
 */
export const forgotPassword = async (e, setValidated) => {
  const form = e.currentTarget;
  if (form.checkValidity() === false) {
    e.preventDefault();
    e.stopPropagation();
  }
  setValidated(true);

  try {
    await axios.post("http://192.168.1.128:9000/recoverPassword", {
      email: email.value,
    });
    window.location.replace("http://192.168.1.128:3000/SentEmail");
  } catch (error) {
    if (error.response) {
      setMsg(error.response.data.msg);
    }
  }
};
